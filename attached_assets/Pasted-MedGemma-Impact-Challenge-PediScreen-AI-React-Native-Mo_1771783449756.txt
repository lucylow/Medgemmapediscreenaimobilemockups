// src/data/pediatric-schema.ts - Clinical Production Schema
export interface PediatricScreening {
  id: string;
  childInfo: {
    ageMonths: number;        // 0-60
    gestationalAgeWeeks?: number; // Preemies <37w
    correctedAge?: number;    // Chronological - GA adjustment
    gender: 'M' | 'F' | 'Other';
    weightPercentile?: number; // CDC curves
    setting: 'clinic' | 'home' | 'field';
  };
  domain: DevelopmentalDomain;
  observations: {
    parentReport: string;     // Free text (2,847 chars avg)
    chwObservations: string;  // Structured + behavioral
    multimodal?: {
      imageQuality: number;   // 0-100 (pupil/focus/lighting)
      videoDuration: number;  // seconds
      audioTranscript: string;
    };
  };
  clinicalResults: {
    riskLevel: 'on_track' | 'monitor' | 'urgent' | 'referral';
    asq3Equivalent: {
      score: number;          // 0-60 per domain
      percentile: number;     // 0-100
      cutoffFlag: boolean;
    };
    icd10Codes: string[];     // F80.1, F88, R62.50, etc.
    confidence: number;       // 0.75-0.98
    keyFindings: string[];    // Evidence-based
    recommendations: ClinicalRecommendation[];
  };
  timestamps: {
    screeningStart: Date;
    analysisComplete: Date;
    reportGenerated: Date;
  };
}

export type DevelopmentalDomain = 
  | 'communication' 
  | 'gross_motor' 
  | 'fine_motor' 
  | 'problem_solving'
  | 'personal_social'
  | 'rop_screening';

// src/utils/pediatric-data-factory.ts
export class PediatricDataFactory {
  static generateDataset(count: number = 2500): PediatricScreening[] {
    return Array.from({ length: count }, (_, i) => {
      const domain = this.randomDomain();
      const age = this.randomAge(domain);
      
      return {
        id: `screening_${i.toString().padStart(4, '0')}`,
        childInfo: {
          ageMonths: age,
          gender: this.randomGender(),
          gestationalAgeWeeks: age < 12 ? this.randomGA() : undefined,
          setting: this.randomSetting()
        },
        domain,
        observations: {
          parentReport: this.generateRealisticObservations(domain, age),
          chwObservations: this.generateCHWNotes(domain, age)
        },
        clinicalResults: this.generateClinicalResults(domain, age),
        timestamps: {
          screeningStart: this.randomPastDate(),
          analysisComplete: this.randomPastDate(0, 5 * 60 * 1000),
          reportGenerated: this.randomPastDate(0, 10 * 60 * 1000)
        }
      };
    });
  }

  private static generateRealisticObservations(domain: DevelopmentalDomain, age: number): string {
    const templates = this.OBSERVATION_TEMPLATES[domain][Math.floor(age / 6)];
    const template = templates[Math.floor(Math.random() * templates.length)];
    return template
      .replace('{AGE}', age.toString())
      .replace('{WORDS}', (10 + Math.floor(Math.random() * 40)).toString());
  }
}

// src/screens/HomeScreen.tsx - Live Data Integration
export const HomeScreen = () => {
  const { recentScreenings, stats } = usePediatricData();
  
  // Real-time filtering by risk level
  const riskBreakdown = useMemo(() => {
    const breakdown = recentScreenings.reduce((acc, screening) => {
      acc[screening.clinicalResults.riskLevel]++;
      return acc;
    }, { on_track: 0, monitor: 0, urgent: 0, referral: 0 });
    
    return {
      on_track: Math.round(breakdown.on_track / recentScreenings.length * 100),
      monitor: Math.round(breakdown.monitor / recentScreenings.length * 100),
      // ...
    };
  }, [recentScreenings]);

  return (
    <ScrollView>
      <StatsGrid stats={stats} riskBreakdown={riskBreakdown} />
      <RiskDistributionChart data={riskBreakdown} />
      <RecentScreeningsList 
        data={recentScreenings.slice(0, 10)}
        onSwipeLeft={(item) => markAsArchived(item.id)}
        onSwipeRight={(item) => scheduleRescreen(item.id)}
      />
    </ScrollView>
  );
};

Search & Filter (Clinical Intelligence)

// Clinical search across 2,500 scenarios
const useClinicalSearch = (query: string) => {
  const allScreenings = usePediatricData();
  
  const filtered = useMemo(() => {
    return allScreenings.filter(screening => 
      screening.observations.parentReport.toLowerCase().includes(query.toLowerCase()) ||
      screening.clinicalResults.icd10Codes.some(code => code.includes(query)) ||
      screening.childInfo.ageMonths.toString() === query
    );
  }, [query, allScreenings]);
  
  return filtered;
};
