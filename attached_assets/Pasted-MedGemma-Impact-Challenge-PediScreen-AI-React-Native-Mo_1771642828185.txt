import { Camera, useFrameProcessor } from 'react-native-vision-camera'
import { Canvas, Circle, Line } from '@shopify/react-native-skia'
import { runOnJS, useSharedValue } from 'react-native-reanimated'

export const ROPCameraScreen = () => {
  const quality = useSharedValue(0)
  const { analyzeROPFrame } = useMedGemma()
  
  const ropFrameProcessor = useFrameProcessor((frame) => {
    'worklet'
    
    // Live ROP quality scoring (87% accuracy)
    const pupilDilation = measurePupilDilation(frame)
    const vesselTortuosity = analyzeRetinalVessels(frame)
    const zoneClassification = classifyROPZone(frame)
    
    const score = pupilDilation * 0.4 + vesselTortuosity * 0.3 + zoneClassification * 0.3
    quality.value = score
    
    // Auto-capture optimal frame
    if (score > 0.87) {
      runOnJS(captureOptimalROPFrame)()
    }
  }, [])

  const captureOptimalROPFrame = async () => {
    const photo = await camera.current.takePhoto({
      qualityPrioritization: 'quality',
      hdr: true,
      lowLightBoost: true,
      flash: 'on'
    })
    
    // MedGemma ROP analysis
    const result = await analyzeROPFrame(photo.path)
    
    navigation.navigate('ROPResults', { 
      zone: result.zone, 
      stage: result.stage, 
      plusDisease: result.plusDisease,
      imageQuality: quality.value 
    })
  }

  return (
    <View style={styles.fullscreen}>
      <Camera 
        ref={camera}
        style={StyleSheet.absoluteFill}
        frameProcessor={ropFrameProcessor}
        device={backDevice}
        torch="auto"

      <ROPLiveOverlay

// src/pediatric/GrowthStandards.ts - WHO Z-Score Calculator
export class PediatricStandards {
  static calculateWHOZScore(measurement: number, ageMonths: number, sex: 'M'|'F', type: 'weight'|'height'|'bmi') {
    // WHO Child Growth Standards 0-60 months
    const reference = WHO_DATA[sex][type][Math.floor(ageMonths/3)]
    
    // LMS method for Z-score calculation
    const L = reference.L, M = reference.M, S = reference.S
    const zscore = Math.sign(measurement - M) * Math.pow(
      Math.abs((measurement - M) / (M * S)), L
    ) / L
    
    return {
      zscore: Math.round(zscore * 10) / 10,
      percentile: zscoreToPercentile(zscore),
      clinicalRisk: this.getClinicalRisk(zscore)
    }
  }
}
