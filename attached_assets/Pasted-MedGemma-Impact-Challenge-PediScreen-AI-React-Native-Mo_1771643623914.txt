// src/qr/QRPatientScanner.tsx - Instant Child Record Access
import { BarCodeScanner } from 'expo-barcode-scanner'
import QRCode from 'react-native-qrcode-svg'
import * as Haptics from 'expo-haptics'
import { useLocalClinicalDB } from '../hooks/useLocalClinicalDB'

export const QRPatientScanner: React.FC = ({ navigation }) => {
  const [hasPermission, setHasPermission] = useState(false)
  const [scannedData, setScannedData] = useState(null)
  const db = useLocalClinicalDB()
  
  const handleBarCodeScanned = async ({ type, data }: any) => {
    Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success)
    
    try {
      // Parse PediScreen QR format
      const patient = await parsePatientQR(data)
      
      // Cache locally (offline-first)
      await db.cachePatient(patient)
      
      // Navigate to screening with patient context
      navigation.navigate('NewScreening', { 
        patientId: patient.studyId,
        demographics: patient 
      })
      
    } catch (error) {
      Haptics.notificationAsync(Haptics.NotificationFeedbackType.Error)
      Alert.alert('Invalid QR Code', 'Please scan a PediScreen patient QR')
    }
  }

  const parsePatientQR = (qrData: string) => {
    // PediScreen QR Schema v2.1
    const parsed = JSON.parse(qrData.replace('pediscreen://patient/', ''))
    
    return {
      studyId: parsed.studyId,
      firstName: parsed.firstName,
      lastName: parsed.lastName,
      dob: new Date(parsed.dob),
      ageMonths: calculateAgeMonths(parsed.dob),
      gestationalAgeWeeks: parsed.ga || 40,
      parentPhone: parsed.parentPhone,
      location: parsed.location,
      riskHistory: parsed.riskHistory || [],
      language: parsed.language || 'en'
    }
  }

  return (
    <View style={styles.scannerContainer}>
      <BarCodeScanner
        onBarCodeScanned={scanned ? undefined : handleBarCodeScanned}
        style={styles.scanner}
        barCodeTypes={[BarCodeScanner.Constants.BarCodeType.qr]}
      />
      
      {/* QR OVERLAY WITH GUIDANCE */}
      <View style={styles.scannerOverlay}>
        <View style={styles.scanFrame} />
        <Text style={styles.scanInstruction}>Hold steady over patient QR</Text>
        <Text style={styles.scanStatus}>
          {scannedData ? 'PATIENT LOADED' : 'SCANNING...'}
        </Text>
      </View>
    </View>
  )
}

// src/qr/QRPatientGenerator.tsx - Create New Patient QR
export const QRPatientGenerator: React.FC = () => {
  const [patientData, setPatientData] = useState({
    firstName: '',
    lastName: '',
    dob: new Date(),
    gestationalAgeWeeks: 40,
    parentPhone: '',
    location: ''
  })
  
  const generatePatientQR = async () => {
    const studyId = `PS${Date.now()}${Math.random().toString(36).substr(2, 6)}`
    
    const qrPayload = {
      protocol: 'pediscreen://patient/',
      studyId,
      firstName: patientData.firstName,
      lastName: patientData.lastName,
      dob: patientData.dob.toISOString(),
      ga: patientData.gestationalAgeWeeks,
      parentPhone: patientData.parentPhone,
      location: patientData.location,
      version: '2.1',
      created: new Date().toISOString()
    }
    
    // Generate QR as SVG + PNG
    const qrCode = await QRCode.toDataURL(JSON.stringify(qrPayload))
    
    // Save to device gallery + share
    await saveToGallery(qrCode, `${studyId}.png`)
    await shareQR(qrCode, `Print & laminate for ${patientData.firstName}`)
    
    // Save patient record
    await db.createPatient({ ...patientData, studyId })
    
    Alert.alert('âœ… Patient QR Generated', 
      `Study ID: ${studyId}\nPrint 3x5cm laminated badge`)
  }
}

    const syncPayload = {
      protocol: 'pediscreen://sync/',
      deviceId: await Device.getIdAsync(),
      pendingRecords: pendingCount,
      lastSync: lastSync?.toISOString() || null,

    return qr
  }
}
