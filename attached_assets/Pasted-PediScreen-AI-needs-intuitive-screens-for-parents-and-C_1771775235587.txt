2. Symptom Selection Screen
src/screens/SymptomSelection.tsx - Categorized symptom browser
typescript
import React, { useState } from 'react';
import { View, Text, FlatList, TouchableOpacity } from 'react-native';
import { useMedGemma } from '../hooks/useMedGemma';
import { styles } from '../styles/symptomStyles';

const symptomCategories = {
  fever: {
    title: 'Fever & Temperature',
    symptoms: ['Fever 100.4¬∞F+', 'Chills', 'Hot/Cold feeling', 'Night sweats']
  },
  respiratory: {
    title: 'Breathing Issues',
    symptoms: ['Fast breathing', 'Wheezing', 'Cough >2hrs', 'Nose flaring']
  },
  feeding: {
    title: 'Feeding Problems',
    symptoms: ['Refusing to eat', 'Vomiting after feeds', 'Diarrhea', 'Less wet diapers']
  },
  behavior: {
    title: 'Behavior Changes',
    symptoms: ['Lethargic', 'Irritable', 'Not consolable', 'Sleepy/unusual']
  }
};

export default function SymptomSelection({ route, navigation }) {
  const { ageGroup } = route.params;
  const { isReady } = useMedGemma();
  const [selectedSymptoms, setSelectedSymptoms] = useState<string[]>([]);

  const selectSymptom = (symptom: string) => {
    const newSelected = selectedSymptoms.includes(symptom)
      ? selectedSymptoms.filter(s => s !== symptom)
      : [...selectedSymptoms, symptom];
    setSelectedSymptoms(newSelected);
  };

  const nextEnabled = selectedSymptoms.length > 0 && isReady;

  return (
    <View style={styles.container}>
      <Text style={styles.header}>Select Symptoms (pick 1-5)</Text>
      <Text style={styles.subheader}>{ageGroup}</Text>

      <FlatList
        data={Object.values(symptomCategories)}
        keyExtractor={(_, i) => `cat-${i}`}
        renderItem={({ item }) => (
          <View style={styles.category}>
            <Text style={styles.categoryTitle}>{item.title}</Text>
            {item.symptoms.map(symptom => (
              <TouchableOpacity
                key={symptom}
                style={[
                  styles.symptomItem,
                  selectedSymptoms.includes(symptom) && styles.selectedSymptom
                ]}
                onPress={() => selectSymptom(symptom)}
              >
                <Text style={styles.symptomText}>{symptom}</Text>
                {selectedSymptoms.includes(symptom) && (
                  <View style={styles.checkmark} />
                )}
              </TouchableOpacity>
            ))}
          </View>
        )}
      />

      <TouchableOpacity
        style={[styles.nextButton, !nextEnabled && styles.disabledButton]}
        onPress={() => navigation.navigate('SymptomDetails', { 
          symptoms: selectedSymptoms, 
          ageGroup 
        })}
        disabled={!nextEnabled}
      >
        <Text style={styles.nextButtonText}>
          Next: Describe Symptoms ({selectedSymptoms.length}/5)
        </Text>
      </TouchableOpacity>
    </View>
  );
}

4. Results Screen with Risk Triage
src/screens/SymptomResults.tsx - Clear action guidance
typescript
import React from 'react';
import { View, Text, ScrollView, TouchableOpacity } from 'react-native';
import { useMedGemma } from '../hooks/useMedGemma';
import { styles } from '../styles/symptomStyles';

export default function SymptomResults({ route }) {
  const { symptoms, ageGroup, duration, severity, details } = route.params;
  const { analyzeScreening, result, isAnalyzing } = useMedGemma();

  React.useEffect(() => {
    analyzeScreening({
      childAgeMonths: parseInt(ageGroup.match(/\d+/)?.[0] || '12'),
      domain: 'symptom_checker',
      observations: `${symptoms.join(', ')}. Duration: ${duration}. Severity: ${severity}/10. Details: ${details}`,
      questionnaire: { general: severity / 10 }
    });
  }, []);

  if (isAnalyzing) {
    return (
      <View style={[styles.container, styles.centered]}>
        <Text style={styles.analyzingText}>Analyzing with MedGemma...</Text>
      </View>
    );
  }

  const riskColor = result?.riskLevel === 'on_track' ? '#34a853' :
                   result?.riskLevel === 'monitor' ? '#fbbc05' :
                   result?.riskLevel === 'discuss' ? '#ff9800' : '#ea4335';

  const triageActions = {
    on_track: ['Continue monitoring at home', 'Routine doctor visit'],
    monitor: ['Call pediatrician today', 'Monitor key symptoms'],
    discuss: ['Contact doctor within 24hrs', 'Watch for worsening'],
    refer: ['Call doctor NOW or go to ER', 'Do not wait']
  };

  return (
    <ScrollView style={styles.container}>
      {/* Risk Banner */}
      <View style={[styles.riskBanner, { backgroundColor: riskColor + '20' }]}>
        <Text style={[styles.riskLevel, { color: riskColor }]}>
          {result?.riskLevel?.toUpperCase() || 'ANALYZING'}
        </Text>
        <Text style={styles.confidence}>
          {Math.round((result?.confidence || 0) * 100)}% Confidence
        </Text>
      </View>

      {/* Summary Cards */}
      <View style={styles.resultCard}>
        <Text style={styles.cardTitle}>What we found</Text>
        <Text style={styles.summary}>{result?.clinicalSummary}</Text>
      </View>

      <View style={styles.resultCard}>
        <Text style={styles.cardTitle}>For Parents</Text>
        <Text style={styles.parentSummary}>{result?.parentSummary}</Text>
      </View>

      {/* Action Steps */}
      <View style={styles.actionSection}>
        <Text style={styles.sectionTitle}>Next Steps</Text>
        {triageActions[result?.riskLevel || 'monitor'].map((action, i) => (
          <TouchableOpacity key={i} style={styles.actionButton}>
            <Text style={styles.actionNumber}>{i + 1}.</Text>
            <Text style={styles.actionText}>{action}</Text>
          </TouchableOpacity>
        ))}
      </View>

      {/* Emergency Hotline */}
      <View style={styles.emergencyCard}>
        <Text style={styles.emergencyTitle}>üö® EMERGENCY?</Text>
        <Text style={styles.emergencyText}>
          Call 911 (US) or your local emergency number immediately if:
        </Text>
        <Text style={styles.criticalSymptoms}>
          ‚Ä¢ Cannot wake child ‚Ä¢ Blue lips ‚Ä¢ Trouble breathing ‚Ä¢ Seizures
        </Text>
      </View>

      <View style={styles.disclaimerSection}>
        <Text style={styles.legalDisclaimer}>
          ‚öñÔ∏è Medical Disclaimer: This AI tool provides guidance only. Always seek professional medical advice.
        </Text>
      </View>
    </ScrollView>
  );
}
